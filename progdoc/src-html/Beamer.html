001/**
002 * A beamer is an item that can be charged with a room and then teleport the
003 * player to that room.
004 * 
005 * @author Sofyan Guillermet-Laouad and AloÃ¯s Fournier 19.05.23
006 */
007public class Beamer extends Item {
008    private Room aChargedRoom;
009    private boolean aIsCharged;
010
011    /**
012     * Constructor of the class Beamer
013     * 
014     * @param pName
015     * @param pDescription
016     * @param pWeight
017     */
018    public Beamer(final String pName, final String pDescription, final double pWeight) {
019        super(pName, pDescription, pWeight);
020        this.aChargedRoom = null;
021        this.aIsCharged = false;
022    }
023
024    /**
025     * Charges the beamer with the room
026     */
027    public void charge(final Room pRoom) {
028        this.aChargedRoom = pRoom;
029        this.aIsCharged = true;
030    }
031
032    /**
033     * Returns the state of the beamer
034     * @return true if the beamer is charged, false otherwise
035     */
036    public boolean isCharged() {
037        return this.aIsCharged;
038    }
039
040    /**
041     * Teleport the player to the charged room
042     * 
043     * @param pPlayer
044     */
045    public void fire(final Player pPlayer) {
046        if (this.aIsCharged && this.aChargedRoom != null) {
047            pPlayer.setCurrentRoom(this.aChargedRoom);
048            this.aIsCharged = false;
049        } else {
050            System.out.println("The beamer is not charged or the saved room is nul.");
051        }
052    }
053}
