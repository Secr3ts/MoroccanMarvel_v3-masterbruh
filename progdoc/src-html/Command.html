003/**
004 * Classe Command - une commande du jeu d'aventure Zuul.
005 *
006 * @author Sofyan Guillermet-Laouad
007 */
008public class Command
009{
010    private String aCommandWord;
011    private String aSecondWord;
012    
013    public Command(final String pFirstWord, final String pSecondWord) {
014        this.aCommandWord = pFirstWord;
015        this.aSecondWord = pSecondWord;
016    }
017    
018    public String getCommandWord() {
019        return this.aCommandWord;
020    }
021    
022    public String getSecondWord() {
023        return this.aSecondWord;
024    }
025    
026    public void setCommandWord(final String pNewCommandWord) {
027        this.aCommandWord = pNewCommandWord;
028    }
029    
030    public void setSecondWord(final String pNewSecondWord) {
031        this.aSecondWord = pNewSecondWord;
032    }
033    
034    public boolean hasSecondWord() {
035        return this.aSecondWord == null ? false : true;
036    }
037    
038    public boolean isUnknown() {
039        return this.aCommandWord == null ? true : false;
040    }
041} // Command
