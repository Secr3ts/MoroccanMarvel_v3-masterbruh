003/**
004 * This class is part of the "World of Zuul" application. 
005 * "World of Zuul" is a very simple, text based adventure game.  
006 * 
007 * This class holds an enumeration table of all command words known to the game.
008 * It is used to recognise commands as they are typed in.
009 *
010 * @author  Michael Kolling and David J. Barnes + D.Bureau
011 * @version 2008.03.30 + 2019.09.25
012 */
013public class CommandWords
014{
015     // a constant array that will hold all valid command words
016    private final String[] aValidCommands;
017    // a constant array that will hold all valid direction words
018    private static final String[] aValidDirection = {"north", "south", "east", "west", "up", "down"};
019    /**
020     * Constructor - initialise the command words.
021     */
022    public CommandWords()
023    {
024        this.aValidCommands = new String[16];
025        this.aValidCommands[0] = "go";
026        this.aValidCommands[1] = "help";
027        this.aValidCommands[2] = "quit";
028        this.aValidCommands[3] = "look";
029        this.aValidCommands[4] = "eat";
030        this.aValidCommands[5] = "back";
031        this.aValidCommands[6] = "take";
032        this.aValidCommands[7] = "drop";
033        this.aValidCommands[8] = "inventory";
034        this.aValidCommands[9] = "charge";
035        this.aValidCommands[10] = "fire";
036        this.aValidCommands[11] = "test";
037        this.aValidCommands[12] = "alea";
038        this.aValidCommands[13] = "items";
039        this.aValidCommands[14] = "1";
040        this.aValidCommands[15] = "2";
041    } // CommandWords()
042    /**
043     * Check whether a given String is a valid direction word.
044     * @param pString
045     * @return true if a given string is a valid direction, false otherwise.
046     */
047    public static boolean isDirection(final String pString) {
048        for (int vI = 0; vI < aValidDirection.length; vI++) {
049            if (aValidDirection[vI].equals(pString)) {
050                return true;
051            }
052        }
053        return false;
054    }
055    /**
056     * Check whether a given String is a valid command word. 
057     * @return true if a given string is a valid command,
058     * false if it isn't.
059     */
060    public boolean isCommand( final String pString )
061    {
062        for ( int vI=0; vI<this.aValidCommands.length; vI++ ) {
063            if ( this.aValidCommands[vI].equals( pString ) )
064                return true;
065        } // for
066        // if we get here, the string was not found in the commands :
067        return false;
068    } // isCommand()
069    /**
070     * Print all valid commands to System.out.
071     */
072    public String getCommandList() {
073        String vCommandList = "";
074        for ( String command : this.aValidCommands ) {
075            vCommandList += command + " ";
076        } // for
077        return vCommandList;
078    } // showAll()
079} // CommandWords
