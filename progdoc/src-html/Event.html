001/**
002 * An even is fired up when you enter certain Rooms, impeaching you to exit while the condition is not met.
003 * @author Sofyan Guillermet-Laouad and AloÃ¯s Fournier 19.05.23
004 */
005public class Event {
006    private String aName;
007    private UserInterface aGUI;
008    private boolean aEventDone = false;
009    private String choice1;
010    private String choice2;
011    private int correctChoice;
012    private Item aItem;
013    private GameEngine aGameEngine;
014    private Flag aFlag;
015    private boolean aRanEvent = false;
016
017    public Event(final UserInterface pGUI, final String pName, final String pChoice1, final String pChoice2) {
018        this(pGUI, pName, pChoice1, pChoice2, 1, null, null);
019    }
020    public Event(final UserInterface pGUI, final String pName, final String pChoice1, final String pChoice2, final int pCorrectChoice, final Item pItem, final Flag pFlag) {
021        this.aGUI = pGUI;
022        this.aName = pName;
023        this.choice1 = pChoice1;
024        this.choice2 = pChoice2;
025        this.correctChoice = pCorrectChoice;
026        this.aItem = pItem;
027        this.aFlag = pFlag;
028    }
029    
030    
031    public void displayChoice() {
032        this.aGUI.println("You have to choose between two options :");
033        this.aGUI.println("1. " + this.choice1);
034        this.aGUI.println("2. " + this.choice2);
035        
036    }
037    
038    public boolean testChoice(final int pChoice) {
039        if (pChoice == this.correctChoice) {
040            return true;
041        } else {
042            return false;        
043        }
044    }
045    
046    public void run(final GameEngine pGameEngine) {
047        if (!aRanEvent) {
048            this.aRanEvent = true;
049            this.aGUI.println("You entered a Room with an event in it, you can not leave until you have completed the event.");
050            this.aGUI.println("You have to find a way to exit the Room.");
051            this.displayChoice();
052            this.aGameEngine = pGameEngine;
053            if(!this.aItem.getItemName().equals("golden-coins")){
054            this.aGameEngine.interpretCommand("take " + this.aItem.getItemName());
055            }
056        }
057    }
058        
059    public boolean isEventRan() {
060        return this.aRanEvent;
061    }
062    
063    public void setEventDone() {
064        this.aEventDone = true;
065        this.aGUI.println("The event has been completed !");
066        this.aGUI.println("Check your bad/good action counter to see if you made the right choice !");
067        if (this.aItem.getItemName().equals("messy-clothes")) {
068            this.aGameEngine.interpretCommand("drop messy-clothes");
069        }
070    }
071
072    public void setGUI(final UserInterface pGUI) {
073        this.aGUI = pGUI;
074    }
075    public boolean isEventDone() {
076        return this.aEventDone;
077    }
078
079    public Item getEventItem() {
080        return this.aItem;
081    }
082
083    public Flag getEventFlag() {
084        return this.aFlag;
085    }
086
087    public void addFlag(final Flag pFlag) {
088        this.aFlag = pFlag;
089    }
090}
