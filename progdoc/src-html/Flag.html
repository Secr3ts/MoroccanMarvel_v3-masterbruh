001/**
002 * Flag class used to track the advancement of the Player
003 */
004public class Flag {
005    private String aFlagName;
006
007    /**
008     * Constructor of the class Flag
009     * @param pFlagName
010     */
011    public Flag(final String pFlagName) {
012        this.aFlagName = pFlagName;
013    }
014
015    /**
016     * Returns the name of the flag
017     * @return the name of the flag
018     */
019    public String getFlagName() {
020        return this.aFlagName;
021    }
022}
