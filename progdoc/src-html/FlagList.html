001import java.util.HashMap;
002
003/**
004 * The FlagList that mutualize the flags of the game
005 * 
006 * @author Sofyan Guillermet-Laouad (with the help of AloÃ¯s Fournier)
007 */
008public class FlagList {
009    private HashMap<String, Flag> aFlags;
010
011    /**
012     * Constructor of the class FlagList
013     */
014    public FlagList() {
015        this.aFlags = new HashMap<>();
016    }
017    /**
018     * Add a flag to the list of flags
019     * 
020     * @param pFlag
021     */
022    public void addFlag(final Flag pFlag) {
023        this.aFlags.put(pFlag.getFlagName(), pFlag);
024    }
025
026    /**
027     * Returns the flag with the given name
028     * 
029     * @param pFlagName
030     * @return the flag with the given name
031     */
032    public Flag getFlag(final String pFlagName) {
033        return this.aFlags.get(pFlagName);
034    }
035
036    /**
037     * Return true if the flag exists, false otherwise
038     * 
039     * @param pFlagName
040     */
041    public boolean hasFlag(final String pFlagName) {
042        return this.aFlags.containsKey(pFlagName);
043    }
044
045    /**
046     * Remove the flag with the given name
047     * 
048     * @param pFlagName
049     */
050    public void removeFlag(final String pFlagName) {
051        this.aFlags.remove(pFlagName);
052    }
053}
