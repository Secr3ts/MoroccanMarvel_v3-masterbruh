003/**
004 *  This class is the main class of the "Moroccan Marvel" application. 
005 *  "Moroccan Marvel" is a very simple, text based adventure game.  Users 
006 *  can walk around some scenery. That's all. It should really be extended 
007 *  to make it more interesting!
008 * 
009 *  To play this game, create an instance of this class.
010 * 
011 *  This main class creates the necessary implementation objects and starts the game off.
012 * 
013 * @author  Michael Kolling and David J. Barnes
014 * @version 2.0 (Jan 2003) DB edited (2019)
015 */
016
017 public class Game
018 {
019     private UserInterface aGUI;
020     private GameEngine aEngine;
021 
022     /**
023      * Create the game and initialise its internal map. Create the inerface and link to it.
024      */
025     public Game() 
026     {
027         this.aEngine = new GameEngine();
028         this.aGUI = new UserInterface( this.aEngine );
029         this.aEngine.setGUI( this.aGUI );
030     }
031
032    /**
033    *  Main play routine.  Loops until end of play.
034    */
035    public static void main(String[] args)
036    {
037        Game vGame = new Game();
038    } // main(.)
039}
