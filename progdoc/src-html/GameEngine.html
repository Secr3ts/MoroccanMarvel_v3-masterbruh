244    //The player eat something
245    private void eat(final Command pCommand) {
246
247        if (!pCommand.hasSecondWord()) {
248            this.aGUI.println("Eat what?");
249            return;
250        }
251
252        String vItemName = pCommand.getSecondWord();
253        if (this.aPlayer.getInventory().hasItem(vItemName) == false) {
254            this.aGUI.println("You don't have that item.");
255            return;
256        }
257
258        Item vItem = this.aPlayer.getInventory().getItem(vItemName);
259        if (this.aPlayer.getInventory().hasItem(vItemName) && vItem.getItemName().equals("golden-coins")) {
260            this.aGUI.println("Are you crazy? You can't eat that!");
261            return;
262        } else if (vItem.getItemName().equals("couscous")) {
263            this.aPlayer.eat(vItemName);
264            this.aGUI.println("Yummy, nothing better than some good food");
265            return;
266        } else if (vItem.getItemName().equals("magic-msemmen")) {
267            this.aPlayer.eat(vItemName);
268            this.aGUI.println("Something mystical is happening to your body");
269            this.aPlayer.setMaxWeight(this.aPlayer.getMaxWeight() * 2);
270            return;
271        } else {
272            this.aPlayer.eat(vItemName);
273        }
274    }

276    /**
277     * Changes the room in which the player is
278     * 
279     * @param pCommand The command
280     */
281    public void goRoom(final Command pCommand) {
282        if (!CommandWords.isDirection(pCommand.getSecondWord())) {
283            this.aGUI.println("Where even is that?");
284            return;
285        }
286
287        if (pCommand.hasSecondWord() == false) {
288            this.aGUI.println("Go Where ?");
289            return;
290        }
291
292        if (this.aPlayer.getCurrentRoom() instanceof TransporterRoom) {
293            TransporterRoom vTransporter = (TransporterRoom) this.aPlayer.getCurrentRoom();
294            if (this.aIsInTestMode && !this.aFakeRandomRoom.isEmpty()) {
295                ArrayList<Room> vRooms = new ArrayList<Room>(vTransporter.getExits());
296                for (Room pRoom : vRooms) {
297                    String vRoomName = pRoom.getDescription().toLowerCase();
298                    if (this.aFakeRandomRoom.contains(vRoomName.toLowerCase())) {
299                        this.aPlayer.setCurrentRoom(pRoom);
300                        this.printLocationInfo();
301                        return;
302                    }
303                }
304                this.printLocationInfo();
305                return;
306            }
307            this.aPlayer.setCurrentRoom(vTransporter.getRandomExit());
308            this.aPlayer.getPreviousRooms().clear();
309            return;
310        }
311
312        if (this.aPlayer.getCurrentRoom().hasEvent()) {
313            if (this.aPlayer.getCurrentRoom().getEvent().isEventRan()) {
314                if (!this.aPlayer.getCurrentRoom().isEventDone()) {
315                    this.aGUI.println("You can't go anywhere until you finish the event.");
316                    return;
317                }
318            }
319        }
320
321        String vDirection = pCommand.getSecondWord();
322        Room vNextRoom = this.aPlayer.getCurrentRoom().getExit(vDirection);
323
324        if (vNextRoom == null) {
325            this.aGUI.println("There is no door !");
326            return;
327        }
328
329        if(vNextRoom.equals(this.aRooms.get("seniorriver2")) && !this.aPlayer.hasFlag("senior")) {
330                this.aGUI.println("There is nothing there... yet?");
331                return;
332            }
333
334        if (this.aPlayer.getCurrentRoom().equals(this.aRooms.get("death"))) {
335                if (this.aPlayer.getBadAction() >= 3) {
336                    if (!vNextRoom.equals(this.aRooms.get("jahannam"))) {
337                        this.aGUI.println("You do not belong there.");
338                        return;
339                    }
340                } else if (this.aPlayer.getGoodAction() >= 3) {
341                    if (!vNextRoom.equals(this.aRooms.get("jannah"))) {
342                        this.aGUI.println("You do not belong there.");
343                        return;
344                    }
345                }
346            }
347
348        this.aPlayer.goRoom(vDirection);
349        if (vNextRoom.hasEvent() && vNextRoom.getEvent().getEventItem().getItemName().equals("old-book")) {
350            if (this.aPlayer.hasFlag("senior")) {
351                this.aPlayer.getCurrentRoom().runEvent(this);
352            }
353        } else if (vNextRoom.hasEvent() && vNextRoom.getEvent().getEventItem().getItemName().equals("golden-coins")
354                && !this.aPlayer.getInventory().hasItem("golden-coins")) {
355            this.aGUI.println("You\'re an adult now, you should probably get to work now");
356        } else {
357            this.aPlayer.getCurrentRoom().runEvent(this);
358        }
359        this.printLocationInfo();
360        if (this.aPlayer.getCurrentRoom().equals(this.aRooms.get("jannah")) || this.aPlayer.getCurrentRoom().equals(this.aRooms.get("jahannam"))) {
361            this.end();
362        } 
363    }

374        } else {
375            this.aGUI.println("You lost the game, you are doomed to eternal suffering in Hell.");
376        }

552    /**
553     * Transform the input into a readable command and interprets it.
554     * 
555     * @param pCommandLine
556     */
557    public void interpretCommand(final String pCommandLine) {
558        this.aGUI.println("> " + pCommandLine);
559        Command vCommand = this.aParser.getCommand(pCommandLine);
560
561        if (vCommand.isUnknown()) {
562            this.aGUI.println("I don't know what you mean...");
563            return;
564        }
565
566        String vCommandWord = vCommand.getCommandWord();
567        vCommandWord = vCommandWord.toLowerCase();
568        switch (vCommandWord) {
569            case "quit":
570                this.endGame();
571                break;
572            case "help":
573                this.printHelp();
574                break;
575            case "look":
576                this.look();
577                break;
578            case "eat":
579                this.eat(vCommand);
580                break;
581            case "test":
582                this.test(vCommand);
583                break;
584            case "back":
585                this.back(vCommand);
586                break;
587            case "take":
588                this.take(vCommand);
589                break;
590            case "drop":
591                this.drop(vCommand);
592                break;
593            case "go":
594                this.goRoom(vCommand);
595                break;
596            case "items":
597                this.printInventory(vCommand);
598                break;
599            case "charge":
600                this.charge(vCommand);
601                break;
602            case "fire":
603                this.fire(vCommand);
604                break;
605            case "alea":
606                this.alea(vCommand);
607                break;
608            case "1":
609                this.choice(vCommand);
610                break;
611            case "2":
612                this.choice(vCommand);
613                break;
614            default:
615                this.aGUI.println("I don't know what you mean...");
616                break;
617        }
618    }

623            return;
624        }
625
626        if (!this.aPlayer.getCurrentRoom().hasEvent()) {
627            this.aGUI.println("There is no event to choose from.");
628            return;
629        }
630
631        if (this.aPlayer.getCurrentRoom().getEvent().testChoice(Integer.parseInt(pCommand.getCommandWord()))) {
632            this.aPlayer.getCurrentRoom().getEvent().setEventDone();
633            if (this.aPlayer.getCurrentRoom().getEvent().getEventItem().getItemName().equals("wooden-toy")) {
634                this.aPlayer.getInventory().removeItem("wooden-toy");
635            } else if (this.aPlayer.getCurrentRoom().getEvent().getEventItem().getItemName().equals("messy-clothes")) {
636                this.interpretCommand("drop messy-clothes");
637            } else if (this.aPlayer.getCurrentRoom().getEvent().getEventItem().getItemName().equals("water-jar")) {
638                this.aPlayer.getInventory().removeItem("water-jar");
639            } else if (this.aPlayer.getCurrentRoom().getEvent().getEventItem().getItemName().equals("golden-coins")) {
640                this.aPlayer.getInventory().getItem("golden-coins").setItemWeight(0.09);
641            }
642            this.aPlayer.addGoodAction();
643        } else {
644            this.aPlayer.getCurrentRoom().getEvent().setEventDone();
645            this.aPlayer.addBadAction();
646        }
647
648        this.aPlayer.addFlag(this.aPlayer.getCurrentRoom().getEvent().getEventFlag());
649    }
