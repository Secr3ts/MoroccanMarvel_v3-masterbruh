001import java.util.HashMap;
002import java.util.Set;
003
004public class ItemList{
005    private HashMap<String, Item> aItemList;
006    private String aLocation;
007
008    public ItemList(String pLocation){
009        this.aItemList = new HashMap<String, Item>();
010        this.aLocation = pLocation;
011    }
012
013    public void addItem(final String pItemName, final Item pItem){
014        this.aItemList.put(pItemName, pItem);
015    }
016
017    public Item getItem(final String pItemName){
018        return this.aItemList.get(pItemName);
019    }
020
021    public void removeItem(final String pItemName){
022        this.aItemList.remove(pItemName);
023    }
024
025    public boolean hasItem(final String pItemName){
026        if(this.aItemList.containsKey(pItemName)){
027            return true;
028        } else{
029            return false;
030        }
031    }
032
033    public String getItemString() {
034        if(this.aItemList.isEmpty()){
035            return "There is nothing here in " + this.aLocation + ".";  
036        }
037
038        StringBuilder vItems = new StringBuilder("Items in " + this.aLocation + ":");
039        Set<String> vKeys = this.aItemList.keySet();
040        for(String vItem : vKeys) {
041            vItems.append(" " + vItem);
042        }
043        return vItems.toString();
044    }//getItemString()
045
046}
