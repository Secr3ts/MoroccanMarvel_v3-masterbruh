001import java.util.ArrayList;
002
003/**
004 * TransporterRoom is a type of room that teleports the player to a random room
005 */
006public class TransporterRoom extends Room {
007
008    private RoomRandormizer aRandomizer;
009
010    /**
011     * Constructor of the class TransporterRoom
012     * @param pName the name of the room
013     * @param pImage the image of the room 
014     */
015    public TransporterRoom(final String pName, final String pImage) {
016        super(pName, pImage);
017        this.aRandomizer = new RoomRandormizer();
018    }
019
020    /**
021     * Add a room to the list of rooms
022     * 
023     * @param pRoom the room to add
024     */
025    public void addRoom(final Room pRoom) {
026        this.aRandomizer.addRoom(pRoom);
027    }
028
029    /**
030     * Returns the list of rooms
031     * 
032     * @return the list of rooms
033     */
034    public ArrayList<Room> getExits() {
035        return new ArrayList<Room>(this.aRandomizer.getRooms());
036    }
037
038    /**
039     * Returns a random room from the list of rooms
040     * 
041     * @return a random room
042     */
043    public Room getRandomExit() {
044        return this.aRandomizer.getRandomRoom();
045    }
046}
